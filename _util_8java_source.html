146  private static Object getJsonSafeObject(Object o) {
147  if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.KITKAT) {
148  return o;
149  } else if (o == null) {
150  return JSONObject.NULL;
151  } else if (o instanceof JSONObject || o instanceof JSONArray) {
152  return o;
153  } else if (o instanceof Collection) {
154  JSONArray retArray = new JSONArray();
155  for (Object entry : (Collection) o) {
156  retArray.put(getJsonSafeObject(entry));
157  }
158  return retArray;
159  } else if (o.getClass().isArray()) {
160  JSONArray retArray = new JSONArray();
161  int length = Array.getLength(o);
162  for (int i = 0; i < length; i++) {
163  retArray.put(getJsonSafeObject(Array.get(o, i)));
164  }
165  return retArray;
166  } else if (o instanceof Map) {
167  return mapToJSONObject((Map)o);
168  } else if (o instanceof Boolean ||
169  o instanceof Byte ||
170  o instanceof Character ||
171  o instanceof Double ||
172  o instanceof Float ||
173  o instanceof Integer ||
174  o instanceof Long ||
175  o instanceof Short ||
176  o instanceof String) {
177  return o;
178  } else if (o.getClass().getPackage().getName().startsWith("java.")) {
179  return o.toString();
180  }
181  return null;
182  }

190  public static long getUTF8Length(String s) {
191  long len = 0;
192  for (int i = 0; i < s.length(); i++) {
193  char code = s.charAt(i);
194  if (code <= 0x7f) {
195  len += 1;
196  } else if (code <= 0x7ff) {
197  len += 2;
198  } else if (code >= 0xd800 && code <= 0xdfff) {
199  // Surrogate pair: These take 4 bytes in UTF-8 and 2 chars in UCS-2
200  // (Assume next char is the other [valid] half and just skip it)
201  len += 4; i++;
202  } else if (code < 0xffff) {
203  len += 3;
204  } else {
205  len += 4;
206  }
207  }
208  return len;
209  }
