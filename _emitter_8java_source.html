380  boolean shutdown(long timeout) {
381  Logger.d(TAG, "Shutting down emitter.");
382  isRunning.compareAndSet(true, false);
383  ExecutorService es = Executor.shutdown();
384  if (es == null || timeout <= 0) {
385  return true;
386  }
387  try {
388  boolean isTerminated = es.awaitTermination(timeout, TimeUnit.SECONDS);
389  Logger.d(TAG, "Executor is terminated: " + isTerminated);
390  return isTerminated;
391  } catch (InterruptedException e) {
392  Logger.e(TAG, "Executor termination is interrupted: " + e.getMessage());
393  return false;
394  }

411  @SuppressWarnings("all")
412  private void attemptEmit() {
413  if (!Util.isOnline(this.context)) {
414  Logger.d(TAG, "Emitter loop stopping: emitter offline.");
415  isRunning.compareAndSet(true, false);
416  return;
417  }
418  if (eventStore.getSize() <= 0) {
419  if (emptyCount >= this.emptyLimit) {
420  Logger.d(TAG, "Emitter loop stopping: empty limit reached.");
421  isRunning.compareAndSet(true, false);
422  return;
423  }
424  emptyCount++;
425  Logger.e(TAG, "Emitter database empty: " + emptyCount);
426  try {
427  this.timeUnit.sleep(this.emitterTick);
428  } catch (InterruptedException e) {
429  Logger.e(TAG, "Emitter thread sleep interrupted: " + e.toString());
430  }
431  attemptEmit();
432  return;
433  }
434  emptyCount = 0;
435 
436  List<EmitterEvent> events = eventStore.getEmittableEvents(sendLimit);
437  List<Request> requests = buildRequests(events);
438  List<RequestResult> results = networkConnection.sendRequests(requests);
439 
440  Logger.v(TAG, "Processing emitter results.");
441 
442  int successCount = 0;
443  int failureCount = 0;
444  List<Long> removableEvents = new ArrayList<>();
445 
446  for (RequestResult res : results) {
447  if (res.getSuccess()) {
448  removableEvents.addAll(res.getEventIds());
449  successCount += res.getEventIds().size();
450  } else {
451  failureCount += res.getEventIds().size();
452  Logger.e(TAG, "Request sending failed but we will retry later.");
453  }
454  }
455  eventStore.removeEvents(removableEvents);
456 
457  Logger.d(TAG, "Success Count: %s", successCount);
458  Logger.d(TAG, "Failure Count: %s", failureCount);
459 
460  if (requestCallback != null) {
461  if (failureCount != 0) {
462  requestCallback.onFailure(successCount, failureCount);
463  } else {
464  requestCallback.onSuccess(successCount);
465  }
466  }
467 
468  if (failureCount > 0 && successCount == 0) {
469  if (Util.isOnline(this.context)) {
470  Logger.e(TAG, "Ensure collector path is valid: %s", getEmitterUri());
471  }
472  Logger.e(TAG, "Emitter loop stopping: failures.");
473  isRunning.compareAndSet(true, false);
474  } else {
475  attemptEmit();
476  }
477  }

489  protected List<Request> buildRequests(List<EmitterEvent> events) {
490  List<Request> requests = new ArrayList<>();
491  String sendingTime = Util.getTimestamp();
492  HttpMethod httpMethod = networkConnection.getHttpMethod();
493 
494  if (httpMethod == GET) {
495  for (EmitterEvent event : events) {
496  Payload payload = event.payload;
497  addSendingTimeToPayload(payload, sendingTime);
498  boolean isOversize = isOversize(payload);
499  Request request = new Request(payload, event.eventId, isOversize);
500  requests.add(request);
501  }
502  } else {
503  for (int i = 0; i < events.size(); i += bufferOption.getCode()) {
504  List<Long> reqEventIds = new ArrayList<>();
505  List<Payload> postPayloadMaps = new ArrayList<>();
506 
507  for (int j = i; j < (i + bufferOption.getCode()) && j < events.size(); j++) {
508  EmitterEvent event = events.get(j);
509  Payload payload = event.payload;
510  Long eventId = event.eventId;
511  addSendingTimeToPayload(payload, sendingTime);
512 
513  if (isOversize(payload)) {
514  Request request = new Request(payload, eventId, true);
515  requests.add(request);
516 
517  } else if (isOversize(payload, postPayloadMaps)) {
518  Request request = new Request(postPayloadMaps, reqEventIds);
519  requests.add(request);
520 
521  // Clear collections and build a new POST
522  postPayloadMaps = new ArrayList<>();
523  reqEventIds = new ArrayList<>();
524 
525  // Build and store the request
526  postPayloadMaps.add(payload);
527  reqEventIds.add(eventId);
528 
529  } else {
530  postPayloadMaps.add(payload);
531  reqEventIds.add(eventId);
532  }
533  }
534 
535  // Check if all payloads have been processed
536  if (!postPayloadMaps.isEmpty()) {
537  Request request = new Request(postPayloadMaps, reqEventIds);
538  requests.add(request);
539  }
540  }
541  }
542  return requests;
543  }

560  private boolean isOversize(@NonNull Payload payload, @NonNull List<Payload> previousPaylods) {
561  long byteLimit = networkConnection.getHttpMethod() == GET ? byteLimitGet : byteLimitPost;
562  return isOversize(payload, byteLimit, previousPaylods);
563  }

572  private boolean isOversize(@NonNull Payload payload, long byteLimit, @NonNull List<Payload> previousPaylods) {
573  long totalByteSize = payload.getByteSize();
574  for (Payload previousPayload : previousPaylods) {
575  totalByteSize += previousPayload.getByteSize();
576  }
577  int wrapperBytes = previousPaylods.size() > 0 ? (previousPaylods.size() + POST_WRAPPER_BYTES) : 0;
578  return totalByteSize + wrapperBytes > byteLimit;
579  }
