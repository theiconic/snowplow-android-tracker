523  public void track(final Event event) {
524  if (!dataCollection.get()) {
525  return;
526  }
527 
528  if (event instanceof ScreenView && screenState != null) {
529  ScreenView screenView = (ScreenView) event;
530  screenView.updateScreenState(screenState);
531  }
532 
533  boolean reportsOnDiagnostic = !(event instanceof TrackerError);
534  Executor.execute(reportsOnDiagnostic, TAG, () -> {
535  event.beginProcessing(this);
536  processEvent(event);
537  event.endProcessing(this);
538  });
539  }

541  private void processEvent(@NonNull Event event) {
542  TrackerEvent trackerEvent = new TrackerEvent(event);
543  Payload payload = payloadWithEvent(trackerEvent);
544  Logger.v(TAG, "Adding new payload to event storage: %s", payload);
545  this.emitter.add(payload);
546  }

548  private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
549  TrackerPayload payload = new TrackerPayload();
550  addBasicPropertiesToPayload(payload, event);
551  if (event.isPrimitive) {
552  addPrimitivePropertiesToPayload(payload, event);
553  } else {
554  addSelfDescribingPropertiesToPayload(payload, event);
555  }
556  List<SelfDescribingJson> contexts = event.contexts;
557  addBasicContextsToContexts(contexts, event);
558  addGlobalContextsToContexts(contexts, payload);
559  wrapContextsToPayload(payload, contexts);
560  return payload;
561  }
562 
563  private void addBasicPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
564  payload.add(Parameters.EID, event.eventId.toString());
565  payload.add(Parameters.DEVICE_TIMESTAMP, Long.toString(event.timestamp));
566  if (event.trueTimestamp != null) {
567  payload.add(Parameters.TRUE_TIMESTAMP, event.trueTimestamp.toString());
568  }
569  payload.add(Parameters.APPID, this.appId);
570  payload.add(Parameters.NAMESPACE, this.namespace);
571  payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
572  if (this.subject != null) {
573  payload.addMap(new HashMap<>(this.subject.getSubject()));
574  }
575  payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
576  }
577 
578  private void addPrimitivePropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
579  payload.add(Parameters.EVENT, event.eventName);
580  payload.addMap(event.payload);
581  }

583  private void addSelfDescribingPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
584  payload.add(Parameters.EVENT, TrackerConstants.EVENT_UNSTRUCTURED);
585  SelfDescribingJson data = new SelfDescribingJson(event.schema, event.payload);
586  HashMap<String, Object> unstructuredEventPayload = new HashMap<>();
587  unstructuredEventPayload.put(Parameters.SCHEMA, TrackerConstants.SCHEMA_UNSTRUCT_EVENT);
588  unstructuredEventPayload.put(Parameters.DATA, data.getMap());
589  payload.addMap(unstructuredEventPayload, base64Encoded, Parameters.UNSTRUCTURED_ENCODED, Parameters.UNSTRUCTURED);
590  }

592  private void addBasicContextsToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull TrackerEvent event) {
593  if (applicationContext) {
594  contexts.add(InstallTracker.getApplicationContext(this.context));
595  }
596 
597  if (mobileContext) {
598  contexts.add(Util.getMobileContext(this.context));
599  }
600 
601  if (event.isService) {
602  return;
603  }
604 
605  if (sessionContext) {
606  String eventId = event.eventId.toString();
607  if (trackerSession.getHasLoadedFromFile()) {
608  synchronized (trackerSession) {
609  SelfDescribingJson sessionContextJson = trackerSession.getSessionContext(eventId);
610  if (sessionContextJson == null) {
611  Logger.track(TAG, "Method getSessionContext method returned null with eventId: %s", eventId);
612  }
613  contexts.add(sessionContextJson);
614  }
615  } else {
616  Logger.track(TAG, "Method getHasLoadedFromFile method returned false with eventId: %s", eventId);
617  }
618  }
619 
620  if (geoLocationContext) {
621  contexts.add(Util.getGeoLocationContext(this.context));
622  }
623 
624  if (screenContext) {
625  contexts.add(screenState.getCurrentScreen(true));
626  }
627 
628  if (gdpr != null) {
629  contexts.add(gdpr.getContext());
630  }
631  }

637  }

641  private void wrapContextsToPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
642  if (contexts.isEmpty()) {
643  return;
644  }
645  List<Map> data = new LinkedList<>();
646  for (SelfDescribingJson context : contexts) {
647  if (context != null) {
648  data.add(context.getMap());
649  }
650  }
651  SelfDescribingJson finalContext = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, data);
652  if (finalContext == null) {
653  return;
654  }
655  payload.addMap(finalContext.getMap(), base64Encoded, Parameters.CONTEXT_ENCODED, Parameters.CONTEXT);
656  }

671  private void addServiceEventPayload(Payload payload, List<SelfDescribingJson> contexts) {
672  // Add default parameters to the payload
673  payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
674  payload.add(Parameters.APPID, this.appId);
675  payload.add(Parameters.NAMESPACE, this.namespace);
676  payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
677 
678  // If there is a subject present for the Tracker add it
679  if (this.subject != null) {
680  payload.addMap(new HashMap<>(this.subject.getSubject()));
681  }
682 
683  // Add Mobile Context
684  if (this.mobileContext) {
685  contexts.add(Util.getMobileContext(this.context));
686  }
687 
688  // Add application context
689  if (this.applicationContext) {
690  contexts.add(InstallTracker.getApplicationContext(this.context));
691  }
692 
693  // If there are contexts to nest
694  if (contexts.size() > 0) {
695  List<Map> contextMaps = new LinkedList<>();
696  for (SelfDescribingJson selfDescribingJson : contexts) {
697  if (selfDescribingJson != null) {
698  contextMaps.add(selfDescribingJson.getMap());
699  }
700  }
701  SelfDescribingJson envelope = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, contextMaps);
702  payload.addMap(envelope.getMap(), this.base64Encoded, Parameters.CONTEXT_ENCODED,
703  Parameters.CONTEXT);
704  }
705 
706  // Add this payload to the emitter
707  this.emitter.add(payload);
708  }
