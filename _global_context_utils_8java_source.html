44  String eventType = (String) payload.getMap().get(Parameters.EVENT);

70  static boolean matchSchemaAgainstRuleSet(RuleSet ruleSet, String eventSchema) {
71  ArrayList<String> acceptRules = ruleSet.getAccept();
72  ArrayList<String> rejectRules = ruleSet.getReject();
73  int acceptCount = 0;
74  int rejectCount = 0;
75 
76  for (String rule : acceptRules) {
77  if (matchSchemaAgainstRule(rule, eventSchema)) {

85  }
86  }

88  return (acceptCount > 0 && rejectCount == 0);

91  static boolean matchSchemaAgainstRule(String rule, String eventSchema) {
92  if ((!isValidRule(rule)) || eventSchema == null) {
93  return false;
94  }
95  String[] ruleParts = getUriSubparts(rule);
96  String[] schemaParts = getUriSubparts(eventSchema);
97 
98  if (ruleParts.length > 0 && schemaParts.length > 0) {
99  if (!matchVendor(ruleParts[0], schemaParts[0])) {
100  return false;
101  }
102  for (int i=1; i<=4; i++) {
103  if (!matchPart(ruleParts[i], schemaParts[i])) {
104  return false;
105  }
106  }
107  return true;
108  }
109  return false;
110  }

112  static boolean isValidRule(String rule) {
113  if (rule != null) {
114  String[] ruleParts = getUriSubparts(rule);
115  if (ruleParts.length == 5) {
116  String vendor = ruleParts[0];
117  String[] versionParts = Arrays.copyOfRange(ruleParts, 2, ruleParts.length);

130  static String[] getUriSubparts(String uri) {
131  // split after excluding protocol
132  String[] parts = uri.substring(5).split("/");
133  if (parts.length == 4) {
134  String[] versionParts = parts[3].split("-");
135  if (versionParts.length == 3) {
136  // parts[2] corresponds to the format of the uri, always same, skipped intentionally
137  return new String[]{ parts[0], parts[1], versionParts[0], versionParts[1], versionParts[2] };
138  }
139  return new String[]{};

165  static boolean validateVendorParts(String[] parts) {
166  if (parts[0].equals("*") || parts[1].equals("*")) {
167  return false;
168  }

173  boolean asterisk = false;

227  static String getSchema(TrackerPayload payload) {
228  HashMap<String,Object> trackerLoad = payload.getMap();
229  try {
230  if (trackerLoad.containsKey(Parameters.UNSTRUCTURED)) {
231  String unstructPayload = (String) trackerLoad.get(Parameters.UNSTRUCTURED);
232  return new JSONObject(unstructPayload).getJSONObject("data").getString("schema");
233  } else if (trackerLoad.containsKey(Parameters.UNSTRUCTURED_ENCODED)) {
234  String encodedSchema = (String) (trackerLoad.get(Parameters.UNSTRUCTURED_ENCODED));
235  JSONObject decodedSchema = new JSONObject(new String(Base64.decode(encodedSchema, Base64.DEFAULT)));
236  return decodedSchema.getJSONObject("data").getString("schema");
237  } else { // the case of first class Snowplow events
238  return TrackerConstants.SCHEMA_PAYLOAD_DATA;
239  }
240  } catch (Exception e) {
241  return "";
242  }
