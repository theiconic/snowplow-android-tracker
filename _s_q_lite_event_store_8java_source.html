174  private List<Map<String, Object>> queryDatabase(String query, String orderBy) {
175  List<Map<String, Object>> res = new ArrayList<>();
176  if (isDatabaseOpen()) {
177  Cursor cursor = null;
178  try {
179  cursor = database.query(EventStoreHelper.TABLE_EVENTS, allColumns, query, null, null, null, orderBy);
180  cursor.moveToFirst();
181  while (!cursor.isAfterLast()) {
182  Map<String, Object> eventMetadata = new HashMap<>();
183  eventMetadata.put(EventStoreHelper.METADATA_ID, cursor.getLong(0));
184  eventMetadata.put(EventStoreHelper.METADATA_EVENT_DATA,
185  Util.deserializer(cursor.getBlob(1)));
186  eventMetadata.put(EventStoreHelper.METADATA_DATE_CREATED, cursor.getString(2));
187  cursor.moveToNext();
188  res.add(eventMetadata);
189  }
190  } finally {
191  if (cursor != null) {
192  cursor.close();
193  }
194  }
195  }
196  return res;
197  }

221  @NonNull
222  @Override
223  public List<EmitterEvent> getEmittableEvents(int queryLimit) {
224  if (!isDatabaseOpen()) {
225  return Collections.emptyList();
226  }
227  insertWaitingEventsIfReady();
228 
229  ArrayList<EmitterEvent> events = new ArrayList<>();
230 
231  // FIFO Pattern for sending events
232  for (Map<String, Object> eventMetadata : getDescEventsInRange(queryLimit)) {
233 
234  // Create a TrackerPayload for each event
235  TrackerPayload payload = new TrackerPayload();
236  Map<String, Object> eventData = (Map<String, Object>)
237  eventMetadata.get(EventStoreHelper.METADATA_EVENT_DATA);
238  payload.addMap(eventData);
239 
240  // Create EmitterEvent
241  Long eventId = (Long) eventMetadata.get(EventStoreHelper.METADATA_ID);
242  if (eventId == null) {
243  Logger.e(TAG, "Unable to get ID of an event extracted from the database.");
244  continue;
245  }
246  EmitterEvent event = new EmitterEvent(payload, eventId);
247  events.add(event);
248  }
249  return events;
250  }
