169  public synchronized SelfDescribingJson getSessionContext(String eventId) {
170  Logger.v(TAG, "Getting session context...");
171  if (!hasLoadedFromFile.get() && !waitForSessionFileLoad()) {
172  hasLoadedFromFile.set(true);
173  }
174  if (!isSessionCheckerEnabled) {
175  return new SelfDescribingJson(TrackerConstants.SESSION_SCHEMA, getSessionValues());
176  }
177  synchronized (this) {
178  if (isSessionUpdateEnabled && shouldUpdateSession()) {
179  Logger.d(TAG, "Update session information.");
180  updateSession(eventId);
181  }
182  lastSessionCheck = System.currentTimeMillis();
183  return new SelfDescribingJson(TrackerConstants.SESSION_SCHEMA, getSessionValues());
184  }
185  }

197  private boolean shouldUpdateSession() {
198  if (isNewSession) {
199  return true;
200  }
201  long now = System.currentTimeMillis();
202  long timeout = isBackground.get() ? backgroundTimeout : foregroundTimeout;
203  return now < lastSessionCheck || now - lastSessionCheck > timeout;
204  }

206  private void updateSession(String eventId) {
207  isNewSession = false;
208  firstId = eventId;
209  previousSessionId = this.currentSessionId;
210  currentSessionId = Util.getUUIDString();
211  sessionIndex++;
212 
213  Logger.d(TAG, "Session information is updated:");
214  Logger.d(TAG, " + Session ID: %s", currentSessionId);
215  Logger.d(TAG, " + Previous Session ID: %s", previousSessionId);
216  Logger.d(TAG, " + Session Index: %s", sessionIndex);
217 
218  boolean isBackground = this.isBackground.get();
219 
220  if (isBackground) { // timed out in background
221  this.executeEventCallback(backgroundTimeoutCallback);
222  } else { // timed out in foreground
223  this.executeEventCallback(foregroundTimeoutCallback);
224  }
225 
226  Executor.execute(true, TAG, () -> {
227  SharedPreferences.Editor editor = sharedPreferences.edit();
228  editor.putString(Parameters.SESSION_USER_ID, userId);
229  editor.putString(Parameters.SESSION_ID, currentSessionId);
230  editor.putString(Parameters.SESSION_PREVIOUS_ID, previousSessionId);
231  editor.putInt(Parameters.SESSION_INDEX, sessionIndex);
232  editor.putString(Parameters.SESSION_FIRST_ID, firstId);
233  editor.putString(Parameters.SESSION_STORAGE, sessionStorage);
234  editor.apply();
235  });

248  public void setIsBackground(boolean isBackground) {
249  Logger.d(TAG, "Application is in the background: %s", isBackground);
250 
251  // If we are currently in the background and the new state is
252  // foreground restart session checking
253  boolean currentState = this.isBackground.get();
254  if (currentState && !isBackground) {
255  Logger.d(TAG, "Application moved to foreground, starting session checking...");
256  this.executeEventCallback(this.foregroundTransitionCallback);
257  try {
258  Tracker.instance().resumeSessionChecking();
259  } catch (Exception e) {
260  Logger.e(TAG, "Could not resume checking as tracker not setup. Exception: %s", e);
261  }
262  }
263 
264  if (!currentState && isBackground) {
265  Logger.d(TAG, "Application moved to background");
266  this.executeEventCallback(this.backgroundTransitionCallback);
267  }
268 
269  this.isBackground.set(isBackground);
270  }
